      def to_s
        row = nil
        [@baseline.to_s, @delta.to_s, @diffnode.to_s, @entry.to_s].each do |col|
          if row.nil?
            row = col if col != ""
          else
            row << "#{$;}#{col}" if col != ""
          end
        end
        row.nil? ? "" : row
      end
