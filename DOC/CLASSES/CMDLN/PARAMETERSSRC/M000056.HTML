    def usage
      hint = "\nUsage: " + $0 + " parameters#{@has_argf ? ' file ...' : ''}\n\n"
      @parms.sort.each do |parm|
        hint += sprintf("\t%s|%s %s \n\t\t%s\n\n", parm.long_form, parm.short_form, parm.arg_desc, parm.parm_desc)
      end
      return hint
    end
